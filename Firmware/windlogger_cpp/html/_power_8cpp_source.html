89 void Power::read_rms_value(unsigned char measure_number, unsigned int crossings, unsigned int timeout){
90 
91  unsigned int crossCount = 0; //Used to measure number of times threshold is crossed.
92  unsigned int numberOfSamples = 0; //This is now incremented
93  sumV = 0;
94  sumI = 0;
95  sumP = 0;
96 
97  double SupplyVoltage = 5000; // VCC/2 is present on ADC5
98  //long SupplyVoltage = readVcc();
99  showSupply = SupplyVoltage ;
100 
101  //-------------------------------------------------------------------------------------------------------------------------
102  // 1) Waits for the waveform to be close to 'zero' (mid-scale adc) part in sin curve.
103  //-------------------------------------------------------------------------------------------------------------------------
104  bool st=false; //an indicator to exit the while loop
105 
106  unsigned long start = TCNT3; //millis()-start makes sure it doesnt get stuck in the loop if there is an error.
107 
108  while(st==false) //the while loop...
109  {
110  startV = adc_value(7); //using the voltage waveform
111  if ((startV < (ADC_COUNTS*0.55)) && (startV > (ADC_COUNTS*0.45))) {
112  st=true; //check its within range
113  //FSM::uart0.print("exit while st startV\r\n");
114  }
115  if ((TCNT3-start)>timeout) {
116  st = true;
117  //FSM::uart0.print("exit while st timeout\r\n");
118  }
119  }
120 
121  //-------------------------------------------------------------------------------------------------------------------------
122  // 2) Main measurement loop
123  //-------------------------------------------------------------------------------------------------------------------------
124 
125  start = TCNT3; // read timer3 value (started for anemometer measure)
126 
127  bool exit_by_crosscount = true;
128 
129  bool exit_by_timeout = true;
130 
131  while ((exit_by_crosscount) && (exit_by_timeout))
132  {
133  char temp_char[12];
134 
135  numberOfSamples++; //Count number of times looped.
136  referenceV = Vmeas; //Used for delay/phase compensation
137 
139  Vmeas=adc_value(7); // channel ?? for the voltage compensated with the offset
140  Imeas=adc_value(5); // channel ?? for the current compensated with the offset
141 
142  // display measures
143  FSM::uart0.print(dtostrf(Vmeas,0,3,temp_char));FSM::uart0.print(" ");FSM::uart0.print(dtostrf(Imeas,0,3,temp_char));FSM::uart0.print("\r\n");
144 
145 
146  //-----------------------------------------------------------------------------
147  // B) Apply digital low pass filters to extract the 2.5 V or 1.65 V dc offset,
148  // then subtract this - signal is now centred on 0 counts.
149  //-----------------------------------------------------------------------------
150  v_offset = v_offset + ((Vmeas-v_offset)/1024);
151  v_filtered = Vmeas - v_offset;
152  i_offset = i_offset + ((Imeas-i_offset)/1024);
153  i_filtered = Imeas - i_offset;
154 
155  sumV += v_filtered*v_filtered; //voltage squared
156  sumI += i_filtered*i_filtered; //current squared
157 
158  Vshifted = referenceV + v_phase * (v_filtered - referenceV); //calculates the voltage shift after the measurement
159 
160  sumP += Vshifted*i_filtered; // stores the power data
161 
162  //-----------------------------------------------------------------------------
163  // G) Find the number of times the voltage has crossed the initial voltage
164  // - every 2 crosses we will have sampled 1 wavelength
165  // - so this method allows us to sample an integer number of half wavelengths which increases accuracy
166  //-----------------------------------------------------------------------------
167  lastVCross = checkVCross;
168  if (Vmeas > startV) checkVCross = true;
169  else checkVCross = false;
170  if (numberOfSamples==1) lastVCross = checkVCross;
171 
172  if (lastVCross != checkVCross) crossCount++;
173 
174  exit_by_crosscount = crossCount < crossings;
175 
176  exit_by_timeout = (TCNT3-start)<timeout;
177 
178  }
179 
180  // check exit
181 
182  //if(!exit_by_crosscount) FSM::uart0.print("exit_by_crosscount\r\n");
183  //if(!exit_by_timeout) FSM::uart0.print("exit_by_timeout\r\n");
184 
185  //-------------------------------------------------------------------------------------------------------------------------
186  // 3) Post loop calculations
187  //-------------------------------------------------------------------------------------------------------------------------
188  //Calculation of the root of the mean of the voltage and current squared (rms)
189  //Calibration coefficients applied.
190 
191  double V_RATIO = v_factor *((SupplyVoltage/1000.0) / (ADC_COUNTS));
192 
193  double I_RATIO = i_factor *((SupplyVoltage/1000.0) / (ADC_COUNTS));
194 
195  v_data[measure_number]= sqrt(sumV/numberOfSamples) *V_RATIO; // stores the rms voltage data
196  i_data[measure_number]= sqrt(sumI/numberOfSamples) * I_RATIO; // stores the rms current data
197  p_data[measure_number]= (sumP/numberOfSamples)*V_RATIO* I_RATIO; // stores the power data
198  s_data[measure_number]= v_data[measure_number]*i_data[measure_number];
199  pf_data[measure_number] = p_data[measure_number]/s_data[measure_number];
